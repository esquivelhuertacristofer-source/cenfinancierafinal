## Supabase — SQL Schema

```sql
-- Progreso por actividad
CREATE TABLE user_activity_progress (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  activity_code TEXT NOT NULL,        -- "ACT-P4-3-3-A"
  unit_code TEXT NOT NULL,            -- "P4-3-3"
  level TEXT NOT NULL,                -- "P4"
  completado BOOLEAN DEFAULT FALSE,
  score INTEGER DEFAULT 0,            -- 0-100
  xp_ganado INTEGER DEFAULT 0,
  intentos INTEGER DEFAULT 1,
  ultimo_intento TIMESTAMPTZ DEFAULT NOW(),
  tiempo_segundos INTEGER DEFAULT 0,
  UNIQUE(user_id, activity_code)
);

-- XP y perfil gamificado
ALTER TABLE user_profiles ADD COLUMN IF NOT EXISTS total_xp INTEGER DEFAULT 0;
ALTER TABLE user_profiles ADD COLUMN IF NOT EXISTS nivel_usuario INTEGER DEFAULT 1;
ALTER TABLE user_profiles ADD COLUMN IF NOT EXISTS racha_dias INTEGER DEFAULT 0;
ALTER TABLE user_profiles ADD COLUMN IF NOT EXISTS ultimo_dia_activo DATE;
ALTER TABLE user_profiles ADD COLUMN IF NOT EXISTS insignias TEXT[] DEFAULT '{}';

-- Función: actualizar XP y nivel automáticamente
CREATE OR REPLACE FUNCTION update_user_xp()
RETURNS TRIGGER AS $$
DECLARE
  nuevo_xp INTEGER;
  nuevo_nivel INTEGER;
BEGIN
  SELECT COALESCE(SUM(xp_ganado), 0) INTO nuevo_xp
  FROM user_activity_progress
  WHERE user_id = NEW.user_id;

  nuevo_nivel := CASE
    WHEN nuevo_xp >= 15000 THEN 6
    WHEN nuevo_xp >= 7000  THEN 5
    WHEN nuevo_xp >= 3500  THEN 4
    WHEN nuevo_xp >= 1500  THEN 3
    WHEN nuevo_xp >= 500   THEN 2
    ELSE 1
  END;

  UPDATE user_profiles
  SET total_xp = nuevo_xp, nivel_usuario = nuevo_nivel
  WHERE id = NEW.user_id;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER on_activity_complete
AFTER INSERT OR UPDATE ON user_activity_progress
FOR EACH ROW EXECUTE FUNCTION update_user_xp();

-- RLS
ALTER TABLE user_activity_progress ENABLE ROW LEVEL SECURITY;
CREATE POLICY "users own progress" ON user_activity_progress
  FOR ALL USING (auth.uid() = user_id);
```

```typescript
import { supabase } from './supabase';
import type { ActivityProgress } from '../types/activities';

export async function saveActivityProgress(progress: Omit<ActivityProgress, 'user_id'>) {
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return;

  await supabase.from('user_activity_progress').upsert({
    user_id: user.id,
    ...progress,
    ultimo_intento: new Date().toISOString(),
  }, { onConflict: 'user_id,activity_code' });
}

export async function getUnitProgress(unit_code: string) {
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return null;
  const { data } = await supabase
    .from('user_activity_progress')
    .select('*')
    .eq('user_id', user.id)
    .eq('unit_code', unit_code);
  return data;
}

export function calcularXP(score: number, xp_base: number, racha: number): number {
  const bonus_perfecto = score === 100 ? 100 : 0;
  const bonus_racha = racha >= 7 ? 150 : racha >= 3 ? 50 : 0;
  return Math.round(xp_base * (score / 100)) + bonus_perfecto + bonus_racha;
}
```

```typescript
'use client';
import { useState, useMemo, useCallback } from 'react';

import { CompletionScreen } from './shared/CompletionScreen';
import { ActivityWrapper } from './shared/ActivityWrapper';

interface Props { data: SimulatorActivityData; }

export default function SimulatorActivity({ data }: Props) {
  const initialValues = Object.fromEntries(data.inputs.map(i => [i.id, i.default]));
  const [values, setValues] = useState<Record<string, number>>(initialValues);
  const [completed, setCompleted] = useState(false);
  const [timeSpent, setTimeSpent] = useState(0);

  const resultado = useMemo(() => {
    try {
      // eslint-disable-next-line no-new-func
      return new Function(...Object.keys(values), `return ${data.formula}`)(...Object.values(values));
    } catch { return 0; }
  }, [values, data.formula]);

  const escenarioActivo = data.escenarios.find(e => {
    try {
      // eslint-disable-next-line no-new-func
      return new Function('resultado', ...Object.keys(values), `return ${e.condicion}`)(resultado, ...Object.values(values));
    } catch { return false; }
  });

  const handleComplete = useCallback(() => setCompleted(true), []);

  if (completed) {
    return <CompletionScreen xp={data.xp} titulo={data.titulo} onNext={() => {}} />;
  }

  return (
    <ActivityWrapper data={data} onComplete={handleComplete}>
      <div className="space-y-6">

          <div key={input.id}>
            <label className="flex justify-between text-sm font-semibold text-[#011C40] mb-2">

              <input

            )}

        ))}
```

```typescript
'use client';
import { useState } from 'react';
import type { QuizActivityData } from '@/types/activities';
import { CompletionScreen } from './shared/CompletionScreen';

interface Props { data: QuizActivityData; }

export default function QuizActivity({ data }: Props) {
  const [indice, setIndice] = useState(0);
  const [seleccion, setSeleccion] = useState<number | null>(null);
  const [correctas, setCorrectas] = useState(0);
  const [mostrarExplicacion, setMostrarExplicacion] = useState(false);
  const [terminado, setTerminado] = useState(false);

  const pregunta = data.preguntas[indice];
  const score = Math.round((correctas / data.preguntas.length) * 100);

  const responder = (opcionIdx: number) => {
    if (seleccion !== null) return;
    setSeleccion(opcionIdx);
    if (opcionIdx === pregunta.correcta) setCorrectas(c => c + 1);
    setMostrarExplicacion(true);
  };

  const siguiente = () => {
    if (indice + 1 >= data.preguntas.length) { setTerminado(true); return; }
    setIndice(i => i + 1);
    setSeleccion(null);
    setMostrarExplicacion(false);
  };

  if (terminado) {
    const xpFinal = Math.round(data.xp * (score / 100)) + (score === 100 ? 100 : 0);
    return <CompletionScreen xp={xpFinal} titulo={data.titulo} score={score} onNext={() => {}} />;
  }

  return (
    <div className="space-y-6">
      <div className="flex items-center gap-2 mb-4">
        {data.preguntas.map((_, i) => (
          <div key={i} className={`h-2 flex-1 rounded-full transition-all ${
            i < indice ? 'bg-green-400' : i === indice ? 'bg-[#FF8C00]' : 'bg-[#E2E8F0]'
          }`} />
        ))}
      </div>

      <p className="text-[#011C40] font-bold text-lg leading-snug">{pregunta.texto}</p>

      <div className="space-y-3">
        {pregunta.opciones.map((opcion, i) => (
          <button
            key={i} onClick={() => responder(i)}
            className={`w-full text-left p-4 rounded-2xl border-2 font-semibold transition-all ${
              seleccion === null ? 'border-[#E2E8F0] hover:border-[#FF8C00]' :
              i === pregunta.correcta ? 'border-green-400 bg-green-50 text-green-700' :
              seleccion === i ? 'border-red-400 bg-red-50 text-red-700' :
              'border-[#E2E8F0] opacity-50'
            }`}
          >
            <span className="mr-3 font-black">{['A','B','C','D'][i]}.</span> {opcion}
          </button>
        ))}
      </div>

      {mostrarExplicacion && (
        <div className="bg-blue-50 border-2 border-blue-200 rounded-2xl p-4 text-sm text-blue-700 font-semibold">
          💡 {pregunta.explicacion}
          <button onClick={siguiente}
            className="mt-3 w-full py-2 bg-[#011C40] text-white rounded-xl font-black text-sm">
            {indice + 1 < data.preguntas.length ? 'Siguiente →' : 'Ver resultado'}
          </button>
        </div>
      )}
    </div>
  );
}
```

## Componente compartido — `CompletionScreen.tsx`

```typescript
'use client';
import { useEffect, useState } from 'react';
import { saveActivityProgress } from '@/lib/activities';

interface Props {
  xp: number;
  titulo: string;
  score?: number;
  activityCode?: string;
  unitCode?: string;
  level?: string;
  onNext: () => void;
}

export function CompletionScreen({ xp, titulo, score = 100, activityCode, unitCode, level, onNext }: Props) {
  const [saved, setSaved] = useState(false);

  useEffect(() => {
    if (!saved && activityCode) {
      saveActivityProgress({
        activity_code: activityCode,
        unit_code: unitCode!,
        level: level!,
        completado: true,
        score,
        xp_ganado: xp,
        intentos: 1,
        ultimo_intento: new Date().toISOString(),
        tiempo_segundos: 0,
      });
      setSaved(true);
    }
  }, []);

  return (
    <div className="text-center space-y-6 py-8">
      <div className="text-6xl animate-bounce">
        {score >= 90 ? '🏆' : score >= 70 ? '⭐' : '💪'}
      </div>
      <h2 className="text-2xl font-black text-[#011C40]">
        {score >= 90 ? '¡Excelente!' : score >= 70 ? '¡Bien hecho!' : '¡Completado!'}
      </h2>
      <div className="bg-[#FF8C00] text-white rounded-2xl p-6 inline-block">
        <p className="text-4xl font-black">+{xp} XP</p>
        <p className="text-sm opacity-80 mt-1">{titulo}</p>
      </div>
      {score !== undefined && (
        <p className="text-[#64748B] font-semibold">{score}% de aciertos</p>
      )}
      <button onClick={onNext}
        className="w-full py-4 bg-[#011C40] text-white rounded-2xl font-black text-lg">
        Siguiente actividad →
      </button>
    </div>
  );
}
```

## Checklist de calidad para cada actividad antes de marcarla como done

- [ ] El `code` sigue el formato `ACT-[LEVEL]-[MODULE]-[UNIT]-[A/B]`
- [ ] Existe el archivo JSON en `src/data/actividades/[level]/[unit_code].json`
- [ ] El componente React renderiza correctamente en móvil (375px) y desktop (1280px)
- [ ] El botón de completar llama a `saveActivityProgress()` con todos los campos
- [ ] Se muestra `CompletionScreen` con XP correcto al finalizar
- [ ] La actividad tiene al menos 1 escenario de retroalimentación educativa
- [ ] Para SIMULADOR: la fórmula evalúa sin errores con cualquier valor del rango
- [ ] Para DECIDE: todos los nodos referenciados en `siguiente_nodo` existen
- [ ] Para QUIZ: cada pregunta tiene exactamente 4 opciones y la `correcta` (0-3) es válida
- [ ] Para ARRASTRA: cada item tiene una `categoria_correcta` que existe en `categorias`
- [ ] Tiempo de carga < 2 segundos (lazy load del componente)
- [ ] Funciona sin sesión activa (usuario invitado ve actividad pero no guarda progreso)
